```html
<meta charset="UTF-8" /><!--Just charset-->
<meta name="language" content="English"><!--for language-->
<meta name="viewport" content="width=device-width, initial-scale=1.0, maximum-scale=1"><!--for viewport-->
<meta name="description" content="DESCRIPTION OF PAGE CONTENT"><!--for description of website | upto 160 characters only-->
<meta name="keywords" content="SOME KEYWORDS"><!--for keywords-->
<meta name="theme-color" content="THEME COLOR IN HEX" /><!--for android chrome topbar theme-->
<link rel="shortcut icon" href="https://imlolman.github.io/img/favicon.ico"><!--fevicon -->
<meta property="og:type" content="article | website" /><!--for telling about website-->
<meta property="og:title" content="TITLE OF YOUR POST OR PAGE" /><!--for fb title-->
<meta property="og:description" content="DESCRIPTION OF PAGE CONTENT" /><!--for fb description-->
<meta property="og:image" content="LINK TO THE IMAGE FILE" /><!--for fb image-->
<meta property="og:url" content="PERMALINK" /><!--for fb permanent URL-->
<meta property="og:site_name" content="SITE NAME" /><!--for overall company name-->
<link rel="canonical" href="PERMALINK" /><!--for fixed current page url-->
<meta name="twitter:title" content="TITLE OF POST OR PAGE"><!--for twitter title-->
<meta name="twitter:description" content="DESCRIPTION OF PAGE CONTENT"><!--for twitter description-->
<meta name="twitter:image" content="LINK TO IMAGE"><!--for twitter image-->
<meta name="twitter:creator" content="@USERNAME"><!--twitter username-->
<meta name="robots" content="index, follow"><!--for google crawling-->
```
